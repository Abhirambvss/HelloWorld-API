# Specify the path to your Excel file
$excelFilePath = "C:\Path\to\your\file.xlsx"

# Load the Excel COM object
$excel = New-Object -ComObject Excel.Application
$workbook = $excel.Workbooks.Open($excelFilePath)
$worksheet = $workbook.Sheets.Item(1)

# Find the Used Range
$usedRange = $worksheet.UsedRange

# Remove Duplicates
$usedRange.RemoveDuplicates(1, 1, [Microsoft.Office.Interop.Excel.XlYesNoGuess]::xlYes)

# Save and Close
$workbook.Save()
$workbook.Close()
$excel.Quit()
[System.Runtime.Interopservices.Marshal]::ReleaseComObject($excel) | Out-Null

Write-Host "Duplicates removed from $excelFilePath."

# Clean up
Remove-Variable excel, workbook, worksheet, usedRange
